User Defined

            fungsi ekspresi(dengan nama):
            function fungsi(){
                return 'eksekusi';
            }
            console.log(fungsi)

            fungsi deklarasi(tanpa nama):
            let fungsi2 = function(){
                return 'eksekusi';
            }

            function fungsi3(parameter1, parameter2 = 'nilai default', dst){
                return parameter1, parameter2, dst;
            }
            console.log(fungsi3(argumen1, argumen2, dst));

            function fungsi4(...restParam){
                let total = 0;
                for (let i = 0; i < restParam.length; i++){
                    total += restParam[i];
                }
                return restParam;
            }
            console.log(fungsi4(1,2,3))
  • parameter1 (optional parameter) jika argumennya diisi maka nilainya sesuai isiannya, jika tidak diisi maka nilainya undefined
  • parameter2 (default parameter) jika argumennya diisi maka nilainya sesuai isiannya, jika tidak diisi maka nilainya 'nilai default'
  • rest parameter (...restParam) Argumen yang diisi akan berupa array

Arrow function

            function deklarasi dengan arrow function:
            let fungsi2 = function(){
                return 'eksekusi';
            }
            let fungsi2 = () => {
                return 'eksekusi';
            }
            let fungsi2 = (nama, umur) => {
                return nama, umur;
            }
            jika eksekusinya hanya 'return' maka bisa disederhanakan seperti ini:
            let fungsi2 = () => `eksekusi ${nama}`;
            }
