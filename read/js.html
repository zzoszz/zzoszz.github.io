------ kode faktorial loop
function factorial(value) {
    let result = 1;
    for (let i = 1; i <= value; i++ ) {
        result *=i;
    }
    return result;
}
console.log(factorial(4));             hasil = 24
console.log(1*2*3*4);                  hasil = 24
------ kode factorial recursive
function factorialRecursive(value) {
    if (value === 1) {
        return 1;
    } else {
        return value * factorialRecursive(value -1 )
    }
}
console.log(factorialRecursive(5));     hasil = 120
console.log(1*2*3*4*5)                  hasil = 120
